\begin{itemize}[label={},itemindent=-2em,leftmargin=2em]
\item {\bf Ementa:} Revisão básica de teoria dos conjuntos, técnicas de contagem, modelo probabilístico para um experimento aleatório, espaços de probabilidade, axiomas de Kolmogorov, probabilidade condicional e independência, função de distribuição de variáveis aleatórias discretas.

\item {\bf Objetivos:} Capacitar o aluno na utilização e compreensão da teoria probabilística. Identificar e reconhecer os problemas que necessitam da utilização de distribuições de probabilidade, distribuições condicionadas, variáveis aleatórias e funções de variáveis aleatórias.

\item {\bf Metodologia:} Serão ministradas aulas teóricas expositivas; utilizados recursos visuais; resolu\-ção de exercícios em sala de aula; solicitação de atividades extraclasse.

\item {\bf Habilidades e Competências:} Ao final da disciplina o aluno estará apto a compreender e desenvolver a teoria relacionada às distribuições de probabilidade. O cronograma de aulas encontra-se no Quadro 1.

\item {\bf Avaliação:} Serão realizadas três avaliações ao longo do semestre. Este plano de ensino permite a vinculação (ou até mesmo substituição) de uma destas avaliações por um trabalho e/ou seminário a critério do professor. Estudantes que perderem uma das provas e/ou trabalho(s), sem justificativa legal (comprovada), terá nota zero de acordo com as normas do sistema acadêmico. A avaliação substitutiva versará sobre todo o conteúdo da disciplina e qualquer estudante poderá fazê-la para substituir a nota da avaliação que faltou ou para substituir a menor nota.
\end{itemize}

	\item Modelo Probabilístico para um experimento aleatório
	\begin{enumerate}[label*=\arabic*.]

		\item Evento

	\item Revisão básica de teoria de conjuntos
	\begin{enumerate}[label*=\arabic*.]

		\item Operações entre conjuntos
	\end{enumerate}

	\item Técnicas de Contagem
	\begin{enumerate}[label*=\arabic*.]
		\item Permutação ou Arranjo simples e com repetição
		\item Permutação Circular
		\item Combinação simples e com repetição
	\end{enumerate}
	\vspace{.3cm}
	\item Definições e espaços de Probabilidade
	\begin{enumerate}[label*=\arabic*.]
		\item Classes de Conjuntos: sigma-álgebra
		\item Definições de Probabilidade
		\item Definição frequentista
		\item Definição Geométrica
		\item Definição Clássica
		\item Axiomas de Kolmogorov
	\end{enumerate}
	\vspace{.3cm}
	\item Probabilidade Condicional e Independência
	\begin{enumerate}[label*=\arabic*.]
		\item Probabilidade condicional
		\item Regra do Produto de Probabilidades
		\item Lei da Probabilidade Total
		\item Diagrama de árvore
		\item Partição do espaço amostral
		\item Teorema de Bayes
		\item Independência de dois eventos
		\item Independência de vários eventos
	\end{enumerate}
	\vspace{.3cm}
	\item Variável aleatória discreta
