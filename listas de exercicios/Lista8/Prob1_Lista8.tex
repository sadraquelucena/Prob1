\begin{center} {\bf Lista de Exercícios 8} \end{center}
\vspace{0.8cm}

	\begin{enumerate}[label=8.\arabic*)]
		\item Considere uma urna contendo 3 bolas vermelhas e 5 pretas. Retire 3 bolas, sem reposição, e defina a variável aleatória $X$ igual ao número de bolas pretas. 
		\begin{enumerate}
			\item Obtenha a função de probabilidade de $X$.
			\item Repita o problema anterior, mas considerando extrações com reposição.
		\end{enumerate}
		
		\item Dois dados equilibrados são lançados.
		\begin{enumerate}
			\item Seja $X$: ``o maior valor entre as duas faces voltadas para cima''. Qual a função de probabilidade de $X$?
			\item Determine a função de distribuição acumulada de $X$.
			\item Seja $Y$: ``a soma das duas faces voltadas para cima''. Qual a função de probabilidade de $Y$?
			\item Determine a função de distribuição acumulada de $Y$.
		\end{enumerate}
		
		\item Sabe-se que uma determinada moeda apresenta cara três vezes mais frequentemente que coroa. Essa moeda é jogada três vezes. Seja $X$ o número de caras que aparece. Estabeleça a distribuição de probabilidade de $X$ e também sua função de distribuição acumulada.
		
		\item De um lote que contém 25 peças, das quais 5 são defeituosas, são escolhidas 4 ao acaso. Seja $X$ o número de defeituosas encontradas. Estabeleça a distribuição de probabilidade de $X$ quando:
		\begin{enumerate}
			\item as peças forem escolhidas com reposição.
			\item as peças forem escolhidas sem reposição.
		\end{enumerate}
		
		\item Um carregamento de oito computadores similares para um ponto de venda contém três que apresentam defeitos. Se uma escola faz uma copra aleatória de dois desses computadores, determine a distribuição de probabilidade para o número de defeituosos. % resp. 10/28, 15/28, 3/28
		
		\item Se uma agência de veículos vende 50\% de seu estoque de certo carro importado com sistema de estacionamento automático.
		\begin{enumerate}
			\item Determine a distribuição de probabilidade de carros com sistema de estacionamento automático ente os próximos quatro vendidos pela agência. % resp. 1/16, 1/4, 3/8, 1/4, 1/16
			\item Determine a função de distribuição acumulada. % resp. 1/16, 5/16, 11/16, 15/16, 1
		\end{enumerate}
		
		\item De uma caixa que contém quatro moedas de 10 centavos e duas de 5 centavos, três moedas são selecionadas aleatoriamente sem reposição. Determine a distribuição de probabilidade para a soma $T$ dos valores das três moedas. %resp 1/5, 3/5, 1/5
		
		\item Suponha que uma moeda perfeita é lançada até que apareça cara pela primeira vez. Seja $X$ o número de lançamentos até que isso aconteça. 
		\begin{enumerate}
			\item Obtenha a função de probabilidade de $X$. 
			\item Obtenha a função de distribuição acumulada de $X$. 
			\item Repita o exercício supondo que a probabilidade de cara é $p$, $p\neq 1/2$.
		\end{enumerate}
		
		\item Uma moeda perfeita é lançada 4 vezes. Seja $Y$ o número de caras obtidas. 
		\begin{enumerate}
			\item Obtenha a função de probabilidade de $Y$.
			\item Repita o problema anterior, considerando agora que a moeda é viciada, sendo a probabilidade de cara dada por $p$, $0 < p < 1$, $p\neq 1/2$.
		\end{enumerate}
		
		\item Seja uma variável aleatória $X$ com função de probabilidade dada na tabela a seguir:
		\begin{center}
			\begin{tabular}{l|cccccc}
				\hline
				$x$ & 0 & 1 & 2 & 3 & 4 & 5\\
				\hline
				$PX=x))$ & 0 & $p^2$ & $p^2$ & $p$ & $p$ & $p^2$\\
				\hline
			\end{tabular}
		\end{center}
		\begin{enumerate}
			\item Encontre o valor de $p$.
			\item Calcule $P(X\geq 4)$ e $P(X<3)$.
			\item Calcule $P(|X-3|\geq 2)$.
		\end{enumerate}
		
		\item Seja $X$ uma variável aleatória cuja função de probabilidade é dada a seguir:
		\begin{center}
			\begin{tabular}{l|ccccccc}
				\hline
				$x$ & $-2$ & $-1$ & 1 & 2 & 4 & 6 & 7\\
				\hline
				$PX=x))$ & $\frac{1}{14}$ & $\frac{2}{21}$ & $\frac{4}{21}$ & $\frac{1}{14}$ & $\frac{5}{42}$ & $\frac{2}{21}$ & $\frac{5}{14}$\\
				\hline
			\end{tabular}
		\end{center}
		\begin{enumerate}
			\item Calcule $P(X^2>9)$.
			\item Calcule $P(|X|\leq 2)$.
			\item Obtenha a função de distribuição acumulada de $3X$.
			\item Obtenha a função de distribuição acumulada de $X^2$.
		\end{enumerate}
	\end{enumerate}
